   INNER JOIN RANK_USER_WORK_EXPERIENCE w on w.username = i.username
   where	r.VOUCHER_STATUS NOT IN (0, 1, 2, 3, 5, 11);


   SELECT * FROM RANK_APPLY_WORK_EXPERIENCE where WORK_START_DATE>WORK_END_DATE;
